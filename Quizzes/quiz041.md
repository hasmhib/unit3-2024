from kivymd.app import MDApp

class quiz041(MDApp):
    def build(self):
        self.count = 0
        pass

    def button_pressed(self, btn):
        the_tern = self.root.ids.tern
        the_tern.text = "It is X's turn"
        btn.md_bg_color = [1, 0, 0, 1]
        self.count += 1

        if self.count % 2 == 0:
            the_tern = self.root.ids.tern
            the_tern.text = "It is Y's turn"
            btn.md_bg_color = [1, 1, 0, 1]
            btn.text = "*"
            btn.font_size = "36pt"

        else:
            the_tern = self.root.ids.tern
            the_tern.text = "It is X's turn"
            btn.md_bg_color = [1, 0, 0, 1]
            btn.text = "#"
            btn.font_size = "36pt"

test = quiz041()
test.run()
